// ===== ADICIONE ESTES IMPORTS NO TOPO DO ARQUIVO =====
import { DocumentArrowDownIcon, ArrowPathIcon } from '@heroicons/react/24/outline';

// ===== ADICIONE ESSES NOVOS ESTADOS NO COMPONENTE (após os estados existentes) =====
const [retrying, setRetrying] = useState(false);
const [lastSyncErrors, setLastSyncErrors] = useState<string[]>([]);

// ===== ADICIONE ESTAS NOVAS FUNÇÕES (após handleSyncFromSupabase) =====

const handleDownloadLog = async () => {
  try {
    const blob = await dctfService.downloadSyncErrorsLog();
    const url = window.URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `sync-errors-${new Date().toISOString().slice(0,10)}.log`;
    document.body.appendChild(a);
    a.click();
    window.URL.revokeObjectURL(url);
    document.body.removeChild(a);
  } catch (err: any) {
    if (err.response?.status === 404) {
      alert('Nenhum log de erros encontrado. Execute a sincronização primeiro.');
    } else {
      alert('Erro ao baixar log: ' + (err.message || 'Erro desconhecido'));
    }
  }
};

const handleRetrySyncErrors = async () => {
  setRetrying(true);
  setSyncError(null);
  setSyncSuccess(null);
  setSyncProgress(null);

  try {
    const result = await dctfService.retrySyncErrors();
    if (result.success) {
      const message = result.message || 
        `Retry concluído: ${result.data?.inserted || 0} inseridos, ${result.data?.updated || 0} atualizados, ${result.data?.errors || 0} erros`;
      setSyncSuccess(message);
      setSyncProgress(result.data || null);
      if (result.data?.errorLog) {
        setLastSyncErrors(result.data.errorLog);
      }
    } else {
      setSyncError(result.error || 'Erro ao fazer retry de sincronização');
    }
  } catch (err: any) {
    setSyncError(err.response?.data?.error || err.message || 'Erro ao fazer retry');
  } finally {
    setRetrying(false);
  }
};

// ===== ADICIONE ESTE NOVO COMPONENTE (após a função handleRetrySyncErrors) =====

const SyncErrorMonitor = () => {
  if (!syncProgress || syncProgress.errors === 0) return null;

  return (
    <div className="bg-yellow-50 border-2 border-yellow-200 rounded-lg p-4 mt-4">
      <div className="flex items-start">
        <ExclamationTriangleIcon className="h-5 w-5 text-yellow-600 mr-2 mt-0.5" />
        <div className="flex-1">
          <h4 className="font-semibold text-yellow-900 mb-2">
            Monitoramento de Erros
          </h4>
          <div className="text-sm text-yellow-800 mb-3">
            {syncProgress.errors} registro(s) falharam na sincronização
          </div>
          
          <div className="flex flex-wrap gap-2">
            <button
              onClick={handleDownloadLog}
              className="flex items-center gap-2 bg-yellow-600 text-white px-4 py-2 rounded-lg hover:bg-yellow-700 text-sm font-medium"
            >
              <DocumentArrowDownIcon className="h-4 w-4" />
              Baixar Log de Erros
            </button>
            
            <button
              onClick={handleRetrySyncErrors}
              disabled={retrying}
              className="flex items-center gap-2 bg-blue-600 text-white px-4 py-2 rounded-lg hover:bg-blue-700 disabled:opacity-50 disabled:cursor-not-allowed text-sm font-medium"
            >
              <ArrowPathIcon className={`h-4 w-4 ${retrying ? 'animate-spin' : ''}`} />
              {retrying ? 'Tentando novamente...' : 'Tentar Novamente'}
            </button>
          </div>

          {lastSyncErrors.length > 0 && (
            <details className="mt-3">
              <summary className="text-sm font-medium text-yellow-900 cursor-pointer hover:text-yellow-700">
                Ver Últimos Erros ({lastSyncErrors.length})
              </summary>
              <div className="mt-2 bg-white rounded border border-yellow-300 p-2 max-h-40 overflow-y-auto">
                <pre className="text-xs text-gray-700 whitespace-pre-wrap font-mono">
                  {lastSyncErrors.slice(0, 10).join('\n\n')}
                  {lastSyncErrors.length > 10 && '\n\n... e mais ' + (lastSyncErrors.length - 10) + ' erros'}
                </pre>
              </div>
            </details>
          )}
        </div>
      </div>
    </div>
  );
};

// ===== NO JSX, LOGO APÓS O BLOCO {syncProgress && (...)} ADICIONE: =====
<SyncErrorMonitor />
